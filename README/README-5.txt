    @staticmethod
    def sex_norminal(this) -> object:
        # male = 0, female = 1
        combine = [this.train, this.test] # train과 test 가 묶입니다. 
        sex_mapping = {'male':0, 'female':1}
        for dataset in combine:
            dataset['Sex'] = dataset['Sex'].map(sex_mapping)
        this.train = this.train # overriding
        this.test = this.test
        return this
